    match /responsaveis/{responsavelId} {
      function donoResponsavel() {
        return signedIn() && request.auth.uid == responsavelId;
      }

      function textoAte(valor, limite) {
        return valor is string && valor.size() <= limite;
      }

      function textoObrigatorioAte(valor, limite) {
        return valor is string && valor.size() > 0 && valor.size() <= limite;
      }

      function criadoEmValido(dados) {
        return !('criadoEm' in dados) || dados.criadoEm is timestamp;
      }

      function consentimentoPrivacidadeValido(consentimento) {
        return consentimento is map &&
          consentimento.keys().hasOnly(['versao', 'escopo', 'aceitoEm']) &&
          consentimento.keys().hasAll(['versao', 'escopo', 'aceitoEm']) &&
          consentimento.versao == 1 &&
          consentimento.escopo == 'uso-alfabetizacao-tea-v1' &&
          consentimento.aceitoEm is timestamp;
      }

      function responsavelValido(dados) {
        return dados.keys().hasOnly([
            'nome',
            'email',
            'consentimentoPrivacidade',
            'criadoEm',
          ]) &&
          dados.keys().hasAll(['nome', 'email', 'consentimentoPrivacidade']) &&
          textoObrigatorioAte(dados.nome, 80) &&
          textoObrigatorioAte(dados.email, 160) &&
          consentimentoPrivacidadeValido(dados.consentimentoPrivacidade) &&
          criadoEmValido(dados);
      }

      function preferenciasValidas(preferencias) {
        return preferencias is map &&
          preferencias.keys().hasOnly([
            'som',
            'animacoes',
            'altoContraste',
            'alvosMaiores',
            'tamanhoFonte',
          ]) &&
          preferencias.keys().hasAll([
            'som',
            'animacoes',
            'altoContraste',
            'alvosMaiores',
            'tamanhoFonte',
          ]) &&
          preferencias.som is bool &&
          preferencias.animacoes is bool &&
          preferencias.altoContraste is bool &&
          preferencias.alvosMaiores is bool &&
          preferencias.tamanhoFonte in ['normal', 'grande', 'extra-grande'];
      }

      function planoIndividualValido(plano) {
        return plano is map &&
          plano.keys().hasOnly([
            'metaAtual',
            'apoioPreferencial',
            'observacaoMediador',
          ]) &&
          plano.keys().hasAll([
            'metaAtual',
            'apoioPreferencial',
            'observacaoMediador',
          ]) &&
          textoAte(plano.metaAtual, 160) &&
          plano.apoioPreferencial in ['visual', 'verbal', 'gestual', 'pausa'] &&
          textoAte(plano.observacaoMediador, 240);
      }

      function cartaoComunicacaoValido(cartao, idEsperado) {
        return cartao is map &&
          cartao.keys().hasOnly(['id', 'rotulo', 'fala', 'apoio']) &&
          cartao.keys().hasAll(['id', 'rotulo', 'fala', 'apoio']) &&
          cartao.id == idEsperado &&
          textoObrigatorioAte(cartao.rotulo, 24) &&
          textoObrigatorioAte(cartao.fala, 90) &&
          textoObrigatorioAte(cartao.apoio, 140);
      }

      function cartoesComunicacaoValidos(cartoes) {
        return cartoes is list &&
          cartoes.size() == 4 &&
          cartaoComunicacaoValido(cartoes[0], 'pausa') &&
          cartaoComunicacaoValido(cartoes[1], 'ajuda') &&
          cartaoComunicacaoValido(cartoes[2], 'nao-sei') &&
          cartaoComunicacaoValido(cartoes[3], 'pronto');
      }

      function planoRegulacaoValido(planoRegulacao) {
        return planoRegulacao is map &&
          planoRegulacao.keys().hasOnly([
            'sinaisPausa',
            'estrategiasAjudam',
            'evitarDuranteSobrecarga',
          ]) &&
          planoRegulacao.keys().hasAll([
            'sinaisPausa',
            'estrategiasAjudam',
            'evitarDuranteSobrecarga',
          ]) &&
          textoAte(planoRegulacao.sinaisPausa, 140) &&
          textoAte(planoRegulacao.estrategiasAjudam, 140) &&
          textoAte(planoRegulacao.evitarDuranteSobrecarga, 140);
      }

      function perfilApoioValido(perfilApoio) {
        return perfilApoio is map &&
          perfilApoio.keys().hasOnly([
            'comunicacaoPreferencial',
            'acessoPreferencial',
            'regulacaoPreferencial',
            'limiteTentativasAntesPausa',
            'cartoesComunicacao',
            'planoRegulacao',
            'observacoes',
          ]) &&
          perfilApoio.keys().hasAll([
            'comunicacaoPreferencial',
            'acessoPreferencial',
            'regulacaoPreferencial',
            'limiteTentativasAntesPausa',
            'cartoesComunicacao',
            'planoRegulacao',
            'observacoes',
          ]) &&
          perfilApoio.comunicacaoPreferencial in [
            'fala',
            'gestos',
            'figuras',
            'dispositivo',
            'emergente',
          ] &&
          perfilApoio.acessoPreferencial in [
            'toque-direto',
            'toque-com-confirmacao',
            'toque-com-ajuda',
            'mouse-teclado',
            'escolha-mediada',
          ] &&
          perfilApoio.regulacaoPreferencial in [
            'pausa',
            'ambiente-calmo',
            'movimento',
            'alternar',
          ] &&
          perfilApoio.limiteTentativasAntesPausa is int &&
          perfilApoio.limiteTentativasAntesPausa >= 2 &&
          perfilApoio.limiteTentativasAntesPausa <= 20 &&
          cartoesComunicacaoValidos(perfilApoio.cartoesComunicacao) &&
          planoRegulacaoValido(perfilApoio.planoRegulacao) &&
          textoAte(perfilApoio.observacoes, 240);
      }

      function atividadesDominadasValidas(atividadesDominadas) {
        return atividadesDominadas is list && atividadesDominadas.size() <= 300;
      }

      // E-mail de colaborador (segundo responsável/terapeuta): validação de
      // formato simples, tamanho igual ao limite de e-mail do responsável.
      // Compartilhamento é só leitura + registrar observação de sessão (ver
      // colaboradorAutorizado abaixo) — nunca editar perfil, criar tentativa,
      // apagar dados ou ver o documento raiz do responsável.
      function colaboradorEmailValido(email) {
        return email is string && email.size() > 0 && email.size() <= 160 &&
          email.matches('.*@.*');
      }

      function colaboradoresEmailValidos(lista) {
        return lista is list && lista.size() <= 3 &&
          (lista.size() < 1 || colaboradorEmailValido(lista[0])) &&
          (lista.size() < 2 || colaboradorEmailValido(lista[1])) &&
          (lista.size() < 3 || colaboradorEmailValido(lista[2]));
      }

      // Um colaborador só é reconhecido se o e-mail do próprio token de auth
      // (não um valor arbitrário) estiver na lista do perfil E esse e-mail já
      // foi verificado — sem isso, alguém poderia se cadastrar com o e-mail de
      // outra pessoa (sem provar posse da caixa de entrada) e ganhar acesso
      // antes do dono legítimo verificar a própria conta.
      function colaboradorAutorizado(dadosPerfil) {
        return signedIn() &&
          request.auth.token.email != null &&
          request.auth.token.email_verified == true &&
          'colaboradoresEmail' in dadosPerfil &&
          dadosPerfil.colaboradoresEmail is list &&
          request.auth.token.email in dadosPerfil.colaboradoresEmail;
      }
